sendToKoodous

Pequeño trozo de codigo que nos permite cargar una apk maliciosa a koodous.com para su posterior análisis



#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import requests
import hashlib
import sys;
filename = sys.argv[1];
print("\nfile: "+filename);
sha256_hash = hashlib.sha256()
files = {'upload_file': open(filename,'rb')}
with open(filename,"rb") as f:
	# Read and update hash string value in blocks of 4K
	for byte_block in iter(lambda: f.read(4096),b""):
		sha256_hash.update(byte_block)
	sha256=sha256_hash.hexdigest();
	print("SHA256: "+sha256);
url_koodous = "https://api.koodous.com/apks/%s/get_upload_url" % sha256
r = requests.get(url=url_koodous, headers={'Authorization': 'Token **TOKENHERE**'});
code=r.status_code;
print("\n"+"="*75+"\nRESPONSE:\n"+"="*75+"\n");
if(code==409):
	print("Ya se cargo antes este archivo");
if(code==200):
	requests.post(url=r.json().get('upload_url'), files=files);
else:
	print("Error: "+str(code));
