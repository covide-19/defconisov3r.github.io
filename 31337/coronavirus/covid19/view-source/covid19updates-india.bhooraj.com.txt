      window.onload = function calC(argument) {
        var x = document.getElementsByTagName("table")[0].rows, totalCases = 0, totalRecovered = 0, totalDeaths = 0 ;
        var y = x.length;
        for(i = 1; i < y; i++)
          {
            totalCases += parseInt(x[i].cells[1].innerHTML);
            totalRecovered += parseInt(x[i].cells[2].innerHTML);
            totalDeaths += parseInt(x[i].cells[3].innerHTML);
          }
        document.getElementById("totalcases").innerHTML = "Total Cases &colon;" + totalCases;
        document.getElementById("totalrecovered").innerHTML = "Total Recovered / Discharged &colon;" + totalRecovered;
        document.getElementById("totaldeaths").innerHTML = "Total Deaths &colon;" + totalDeaths;
      }
