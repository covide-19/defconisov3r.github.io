            $(this)
                .data('aspectRatio', (this.height / this.width) || 0.5625)

              var newWidth = $fluidEl.width() - (isBackup ? 200 : 0);
              if (!newWidth) return;
              $allVideos.each(function() {
                var $el = $(this);
                $el
                  .width(newWidth)
                  .height(newWidth * $el.data('aspectRatio'));
              });
          }).resize();
        });
