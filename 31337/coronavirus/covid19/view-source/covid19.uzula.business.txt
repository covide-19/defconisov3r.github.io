    <script type="text/javascript">
      // Google GeoMapの初期設定
      google.charts.load('current', {
        'packages':['geochart'],
        // Note: you will need to get a mapsApiKey for your project.
        // See: https://developers.google.com/chart/interactive/docs/basic_load_libs#load-settings
        'mapsApiKey': 'AIzaSyD-9tSrke72PouQMnMX-a7eZSW0jkFMBWY'
      });
      google.charts.setOnLoadCallback(getLocationChart);

      // 折れ線グラフのインスタンス
      var lineChart = null;

      $(document).ready(function (){
        getLineChart();
        getLocationChart();
      });
      $(document).on('change', '#filterByMonthRange', function (){
        getLineChart();
      });
      $(document).on('change', '#filterByLocation', function (){
        getLineChart();
      });
      $(document).on('change', '#filterByDataType', function (){
        getLocationChart();
      });
      $(document).on('change', '#filterByDatePoint', function (){
        getLocationChart();
      });

      /* 折れ線グラフのデータ取得 */
      function getLineChart() {
        var selectedMonthRange = $('#filterByMonthRange').val();
        var selectedLocation = $('#filterByLocation').val();

        $.ajax({
          url: 'http://covid19.uzula.business/getLineChartData?monthRange=' + selectedMonthRange + '&location=' + selectedLocation,
          type: 'GET',
          async: true,
          dataType: 'json',
          data: { monthRange: selectedMonthRange, location: selectedLocation },
          cache: false,
          //processData: false,
        }).done(function (data) {
          rendarLineChart(data);
        }).fail(function (jqXHR, textStatus, errorThrown) {
          if (jqXHR.status === 401) { location.href = 'http://covid19.uzula.business'; }
          console.log("failed to call api. status:" + jqXHR.status);
        });
      }

      /* 折れ線グラフの描画 */
      function rendarLineChart(datasource) {
        // LineChartのデータセット
        var chartData = {
          labels  : datasource.dateList,
          datasets: [
            {
              label               : 'Total Cases',
              backgroundColor     : 'transparent',
              borderColor         : 'orange',
              pointRadius          : true,
              pointColor          : 'orange',
              data                : datasource.totalCases
            },
            {
              label               : 'Total Deaths',
              backgroundColor     : 'transparent',
              borderColor         : 'red',
              pointRadius          : true,
              pointColor          : 'red',
              data                : datasource.totalDeaths
            },
            {
              label               : 'New Cases',
              backgroundColor     : 'transparent',
              borderColor         : 'rgba(60,141,188,0.8)',
              borderColor         : 'green',
              pointRadius          : true,
              pointColor          : 'green',
              data                : datasource.newCases
            },
            {
              label               : 'New Deaths',
              backgroundColor     : 'transparent',
              borderColor         : 'rgba(60,141,188,0.8)',
              borderColor         : 'purple',
              pointRadius          : true,
              pointColor          : 'purple',
              data                : datasource.newDeaths
            },
          ]
        }

        var chartOptions = {
          maintainAspectRatio : false,
          responsive : true,
          tooltips: {
            mode: 'index',
            intersect: false,
          },
          legend: {
            display: true,
            position: 'top'
          },
          scales: {
            xAxes: [{
              gridLines : {
                display : false,
              }
            }],
            yAxes: [{
              gridLines : {
                display : false,
              }
            }]
          }
        }

        //-------------
        //- LINE CHART -
        //--------------
        var lineChartCanvas = jQuery('#lineChart').get(0).getContext('2d')
        var lineChartOptions = jQuery.extend(true, {}, chartOptions)
        var lineChartData = jQuery.extend(true, {}, chartData)
        lineChartData.datasets[0].fill = false;
        lineChartData.datasets[1].fill = false;
        lineChartOptions.datasetFill = false

        if (lineChart) lineChart.destroy();
        lineChart = new Chart(lineChartCanvas, { 
          type: 'line',
          data: lineChartData, 
          options: lineChartOptions
        })
      }

      /* 世界地図のデータ取得 */
      function getLocationChart() {
        var selectedDatePoint = $('#filterByDatePoint').val();

        $.ajax({
          url: 'http://covid19.uzula.business/getLocationChartData?date=' + selectedDatePoint,
          type: 'GET',
          async: true,
          dataType: 'json',
          data: { date: selectedDatePoint },
          cache: false,
          //processData: false,
        }).done(function (data) {
          rendarLocationChart(data);
        }).fail(function (jqXHR, textStatus, errorThrown) {
          if (jqXHR.status === 401) { location.href = 'http://covid19.uzula.business'; }
          console.log("failed to call api. status:" + jqXHR.status);
        });
      }

      /* 世界地図の描画 */
      function rendarLocationChart(dataSource) {
        var data = {};
        var dataType = $('#filterByDataType').val();
        switch(dataType) {
          case 'total_cases':
            data = google.visualization.arrayToDataTable(dataSource.totalCases);
            break;
          case 'total_deaths':
            data = google.visualization.arrayToDataTable(dataSource.totalDeaths);
            break;
          case 'new_cases':
            data = google.visualization.arrayToDataTable(dataSource.newCases);
            break;
          case 'new_deaths':
            data = google.visualization.arrayToDataTable(dataSource.newDeaths);
            break;
        }

        var options = { };
        var chart = new google.visualization.GeoChart(document.getElementById('location_map'));
        chart.draw(data, options);
      }
